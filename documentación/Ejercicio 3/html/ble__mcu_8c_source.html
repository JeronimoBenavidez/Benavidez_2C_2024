200 static void gap_event_handler(esp_gap_ble_cb_event_t event, esp_ble_gap_cb_param_t *param) {
201  CMD_t cmdBuf;
202  static uint8_t adv_config_done = 0;
203  switch (event) {
204  case ESP_GAP_BLE_SCAN_RSP_DATA_SET_COMPLETE_EVT:
205  adv_config_done &= (~SCAN_RSP_CONFIG_FLAG);
206  if (adv_config_done == 0){
207  esp_ble_gap_start_advertising(&spp_adv_params);
208  status = BLE_DISCONNECTED;
209  }
210  break;
211  case ESP_GAP_BLE_ADV_DATA_SET_COMPLETE_EVT:
212  adv_config_done &= (~ADV_CONFIG_FLAG);
213  if (adv_config_done == 0){
214  esp_ble_gap_start_advertising(&spp_adv_params);
215  status = BLE_DISCONNECTED;
216  }
217  break;
218  case ESP_GAP_BLE_ADV_START_COMPLETE_EVT:
219  //advertising start complete event to indicate advertising start successfully or failed
220  if (param->adv_start_cmpl.status != ESP_BT_STATUS_SUCCESS) {
221  ESP_LOGE(__FUNCTION__, "advertising start failed, error status = %x", param->adv_start_cmpl.status);
222  break;
223  }
224  ESP_LOGI(TAG, "Advertising start");
225  break;
226  case ESP_GAP_BLE_PASSKEY_REQ_EVT: /* passkey request event */
227 
228  break;
229  case ESP_GAP_BLE_OOB_REQ_EVT: {
230  uint8_t tk[16] = {1}; //If you paired with OOB, both devices need to use the same tk
231  esp_ble_oob_req_reply(param->ble_security.ble_req.bd_addr, tk, sizeof(tk));
232  break;
233  }
234  case ESP_GAP_BLE_LOCAL_IR_EVT: /* BLE local IR event */
235 
236  break;
237  case ESP_GAP_BLE_LOCAL_ER_EVT: /* BLE local ER event */
238 
239  break;
240  case ESP_GAP_BLE_NC_REQ_EVT:
241  /* The app will receive this evt when the IO has DisplayYesNO capability and the peer device IO also has DisplayYesNo capability.
242  show the passkey number to the user to confirm it with the number displayed by peer device. */
243  esp_ble_confirm_reply(param->ble_security.ble_req.bd_addr, true);
244  //ESP_LOGI(__FUNCTION__, "ESP_GAP_BLE_NC_REQ_EVT, the passkey Notify number:%"PRIu32, param->ble_security.key_notif.passkey);
245  break;
246  case ESP_GAP_BLE_SEC_REQ_EVT:
247  /* send the positive(true) security response to the peer device to accept the security request.
248  If not accept the security request, should send the security response with negative(false) accept value*/
249  esp_ble_gap_security_rsp(param->ble_security.ble_req.bd_addr, true);
250  break;
251  case ESP_GAP_BLE_PASSKEY_NOTIF_EVT:
252 
253  break;
254  case ESP_GAP_BLE_KEY_EVT:
255 
256  break;
257  case ESP_GAP_BLE_AUTH_CMPL_EVT: {
258  cmdBuf.command = CMD_BLUETOOTH_AUTH;
259  xQueueSend(xQueueEvents, &cmdBuf, 0);
260  break;
261  }
262  case ESP_GAP_BLE_REMOVE_BOND_DEV_COMPLETE_EVT: {
263  ESP_LOGD(__FUNCTION__, "ESP_GAP_BLE_REMOVE_BOND_DEV_COMPLETE_EVT status = %d", param->remove_bond_dev_cmpl.status);
264  ESP_LOGI(__FUNCTION__, "ESP_GAP_BLE_REMOVE_BOND_DEV");
265  ESP_LOGI(__FUNCTION__, "-----ESP_GAP_BLE_REMOVE_BOND_DEV----");
266  esp_log_buffer_hex(__FUNCTION__, (void *)param->remove_bond_dev_cmpl.bd_addr, sizeof(esp_bd_addr_t));
267  ESP_LOGI(__FUNCTION__, "------------------------------------");
268  break;
269  }
270  case ESP_GAP_BLE_SET_LOCAL_PRIVACY_COMPLETE_EVT:
271  if (param->local_privacy_cmpl.status != ESP_BT_STATUS_SUCCESS){
272  ESP_LOGE(__FUNCTION__, "config local privacy failed, error status = %x", param->local_privacy_cmpl.status);
273  break;
274  }
275  esp_err_t ret = esp_ble_gap_config_adv_data(&spp_adv_config);
276  if (ret){
277  ESP_LOGE(__FUNCTION__, "config adv data failed, error code = %x", ret);
278  }else{
279  adv_config_done |= ADV_CONFIG_FLAG;
280  }
281  ret = esp_ble_gap_config_adv_data(&spp_scan_rsp_config);
282  if (ret){
283  ESP_LOGE(__FUNCTION__, "config adv data failed, error code = %x", ret);
284  }else{
285  adv_config_done |= SCAN_RSP_CONFIG_FLAG;
286  }
287  break;
288  default:
289  break;
290  }
291 }

293 static void gatts_profile_event_handler(esp_gatts_cb_event_t event,
294  esp_gatt_if_t gatts_if, esp_ble_gatts_cb_param_t *param) {
295  esp_ble_gatts_cb_param_t *p_data = (esp_ble_gatts_cb_param_t *) param;
296  CMD_t cmdBuf;
297 
298  switch (event) {
299  case ESP_GATTS_REG_EVT:
300  esp_ble_gap_set_device_name(device_name);
301  //generate a resolvable random address
302  esp_ble_gap_config_local_privacy(true);
303  esp_ble_gap_config_adv_data_raw((uint8_t *)spp_adv_data, sizeof(spp_adv_data));
304  esp_ble_gatts_create_attr_tab(spp_gatt_db, gatts_if, SPP_IDX_NB, SPP_SVC_INST_ID);
305  break;
306  case ESP_GATTS_READ_EVT:
307  break;
308  case ESP_GATTS_WRITE_EVT:
309  cmdBuf.command = CMD_BLUETOOTH_DATA;
310  memcpy(cmdBuf.payload, param->write.value, param->write.len);
311  cmdBuf.length = param->write.len;
312  xQueueSend(xQueueRead, &cmdBuf, 0);
313  break;
314  case ESP_GATTS_EXEC_WRITE_EVT:
315  break;
316  case ESP_GATTS_MTU_EVT:
317  break;
318  case ESP_GATTS_CONF_EVT:
319  break;
320  case ESP_GATTS_UNREG_EVT:
321  break;
322  case ESP_GATTS_DELETE_EVT:
323  break;
324  case ESP_GATTS_START_EVT:
325  break;
326  case ESP_GATTS_STOP_EVT:
327  break;
328  case ESP_GATTS_CONNECT_EVT:
329  /* start security connect with peer device when receive the connect event sent by the master */
330  esp_ble_set_encryption(param->connect.remote_bda, ESP_BLE_SEC_ENCRYPT_MITM);
331  cmdBuf.command = CMD_BLUETOOTH_CONNECT;
332  cmdBuf.spp_conn_id = p_data->connect.conn_id;
333  cmdBuf.spp_gatts_if = gatts_if;
334  xQueueSend(xQueueEvents, &cmdBuf, portMAX_DELAY);
335  break;
336  case ESP_GATTS_DISCONNECT_EVT:
337  cmdBuf.command = CMD_BLUETOOTH_DISCONNECT;
338  status = BLE_DISCONNECTED;
339  xQueueSend(xQueueEvents, &cmdBuf, portMAX_DELAY);
340  /* start advertising again when missing the connect */
341  esp_ble_gap_start_advertising(&spp_adv_params);
342  break;
343  case ESP_GATTS_OPEN_EVT:
344  break;
345  case ESP_GATTS_CANCEL_OPEN_EVT:
346  break;
347  case ESP_GATTS_CLOSE_EVT:
348  break;
349  case ESP_GATTS_LISTEN_EVT:
350  break;
351  case ESP_GATTS_CONGEST_EVT:
352  break;
353  case ESP_GATTS_CREAT_ATTR_TAB_EVT: {
354  if (param->create.status == ESP_GATT_OK){
355  if(param->add_attr_tab.num_handle == SPP_IDX_NB) {
356  memcpy(spp_handle_table, param->add_attr_tab.handles,
357  sizeof(spp_handle_table));
358  esp_ble_gatts_start_service(spp_handle_table[SPP_IDX_SVC]);
359  }else{
360  ESP_LOGE(__FUNCTION__, "Create attribute table abnormally, num_handle (%d) doesn't equal to SPP_IDX_NB(%d)",
361  param->add_attr_tab.num_handle, SPP_IDX_NB);
362  }
363  }else{
364  ESP_LOGE(__FUNCTION__, " Create attribute table failed, error code = %x", param->create.status);
365  }
366  break;
367  }
368  default:
369  break;
370  }
371 }

373 static void gatts_event_handler(esp_gatts_cb_event_t event, esp_gatt_if_t gatts_if,
374  esp_ble_gatts_cb_param_t *param) {
375  /* If event is register event, store the gatts_if for each profile */
376  if (event == ESP_GATTS_REG_EVT) {
377  if (param->reg.status == ESP_GATT_OK) {
378  spp_profile_tab[SPP_PROFILE_APP_IDX].gatts_if = gatts_if;
379  } else {
380  return;
381  }
382  }
383  do {
384  int idx;
385  for (idx = 0; idx < SPP_PROFILE_NUM; idx++) {
386  if (gatts_if == ESP_GATT_IF_NONE || /* ESP_GATT_IF_NONE, not specify a certain gatt_if, need to call every profile cb function */
387  gatts_if == spp_profile_tab[idx].gatts_if) {
388  if (spp_profile_tab[idx].gatts_cb) {
389  spp_profile_tab[idx].gatts_cb(event, gatts_if, param);
390  }
391  }
392  }
393  } while (0);
394 }

406 void bluetooth_events_task(void * arg) {
407  CMD_t cmdBuf;
408  uint16_t spp_conn_id = 0xffff;
409  esp_gatt_if_t spp_gatts_if = 0xff;
410  int data_sent, i;
411 
412  while(1){
413  vTaskDelay(50 / portTICK_PERIOD_MS);
414  xQueueReceive(xQueueEvents, &cmdBuf, portMAX_DELAY);
415  switch(cmdBuf.command){
416  case CMD_BLUETOOTH_CONNECT:
417  spp_conn_id = cmdBuf.spp_conn_id;
418  spp_gatts_if = cmdBuf.spp_gatts_if;
419  break;
420  case CMD_BLUETOOTH_AUTH:
421  ESP_LOGI(TAG, "Device connected");
422  status = BLE_CONNECTED;
423  break;
424  case CMD_BLUETOOTH_DISCONNECT:
425  ESP_LOGI(TAG, "Device disconnected");
426  status = BLE_DISCONNECTED;
427  break;
428  case CMD_SEND_DATA:
429  if (status == BLE_CONNECTED) {
430  data_sent = 0;
431  i = 0;
432  while(data_sent < cmdBuf.length){
433  if((cmdBuf.length - data_sent) > MTU_MAX_BYTES){
434  esp_ble_gatts_send_indicate(spp_gatts_if, spp_conn_id, spp_handle_table[SPP_IDX_SPP_DATA_NOTIFY_VAL], MTU_MAX_BYTES, &cmdBuf.payload[i*MTU_MAX_BYTES], false);
435  data_sent += MTU_MAX_BYTES;
436  i++;
437  }else{
438  esp_ble_gatts_send_indicate(spp_gatts_if, spp_conn_id, spp_handle_table[SPP_IDX_SPP_DATA_NOTIFY_VAL], cmdBuf.length - data_sent, &cmdBuf.payload[i*MTU_MAX_BYTES], false);
439  data_sent += MTU_MAX_BYTES;
440  }
441  }
442  }
443  break;
444  case CMD_BLUETOOTH_DATA:
445  xQueueSend(xQueueRead, &cmdBuf, portMAX_DELAY);
446  break;
447  }
448  }

452 void BleInit(ble_config_t * ble_device){
453 esp_err_t ret;
454  device_name = ble_device->device_name;
455  ble_read_isr_p = ble_device->func_p;
456  /* Initialize NVS. */
457  ret = nvs_flash_init();
458  if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
459  ESP_ERROR_CHECK(nvs_flash_erase());
460  ret = nvs_flash_init();
461  }
462  ESP_ERROR_CHECK(ret);
463  ESP_ERROR_CHECK(esp_bt_controller_mem_release(ESP_BT_MODE_CLASSIC_BT));
464  esp_bt_controller_config_t bt_cfg = BT_CONTROLLER_INIT_CONFIG_DEFAULT();
465  ret = esp_bt_controller_init(&bt_cfg);
466  if (ret) {
467  ESP_LOGE(TAG, "%s init controller failed: %s", __func__, esp_err_to_name(ret));
468  return;
469  }
470  ret = esp_bt_controller_enable(ESP_BT_MODE_BLE);
471  if (ret) {
472  ESP_LOGE(TAG, "%s enable controller failed: %s", __func__, esp_err_to_name(ret));
473  return;
474  }
475  ret = esp_bluedroid_init();
476  if (ret) {
477  ESP_LOGE(TAG, "%s init bluetooth failed: %s", __func__, esp_err_to_name(ret));
478  return;
479  }
480  ret = esp_bluedroid_enable();
481  if (ret) {
482  ESP_LOGE(TAG, "%s enable bluetooth failed: %s", __func__, esp_err_to_name(ret));
483  return;
484  }
485  ret = esp_ble_gatts_register_callback(gatts_event_handler);
486  if (ret){
487  ESP_LOGE(TAG, "gatts register error, error code = %x", ret);
488  return;
489  }
490  ret = esp_ble_gap_register_callback(gap_event_handler);
491  if (ret){
492  ESP_LOGE(TAG, "gap register error, error code = %x", ret);
